question = "How is an AI voice agent different from an IVR system?"
answer = "Traditional IVR systems force callers through rigid \"Press 1 for sales\" menu trees. AI voice agents understand natural speech, interpret intent, and hold dynamic conversations. They adapt to what the caller says rather than forcing them to follow predefined paths."

[[extra.faq_items]]
question = "Will the voice agent sound robotic?"
answer = "Modern AI voice agents use advanced text-to-speech models (ElevenLabs, Cartesia) and voice cloning to produce natural-sounding speech with proper tone, pacing, and emotion. Combined with turn detection and backchanneling (&ldquo;uh-huh,&rdquo; &ldquo;I see&rdquo;), conversations feel natural and human-like."

[[extra.faq_items]]
question = "Can the agent transfer to a human when needed?"
answer = "Yes. AI voice agents can perform warm transfers — they brief the human agent on the conversation context before handing off, so the caller never has to repeat themselves. You define the conditions for escalation based on intent, sentiment, or complexity."

[[extra.faq_items]]
